#include <tunables/global>

  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within the app
  /data/** rw,

  # Docker socket access (requires Protection Mode disabled)
  /var/run/docker.sock rw,

  # nginx
  /usr/sbin/nginx ix,
  /etc/nginx/** r,
  /var/log/nginx/** rw,
  /var/lib/nginx/** rw,
  /run/nginx.pid rw,

  # Node.js and Dockhand application
  /usr/bin/node ix,
  /app/** r,
  /home/dockhand/** rw,

  # NodeSource node_modules
  /usr/lib/node_modules/** r,
}
